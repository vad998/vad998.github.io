Скрипт:
let secCount = 0;
let addCount = 0;
let mpyCount = 0;
let pwrCount = 0;
const sec = a => {
secCount++
return 1 + a
};
const add = (a, b) => {
addCount++
return (b === 0) ? a : sec(add(a, b - 1))
};
const mpy = (a, b) => {
mpyCount++
return (b === 1) ? a : add(a, mpy(a, b - 1))
};
// mpy(11,3) = 11 + mpy(11,2) = 11 + 11 + mpy(11,1) = 11 + 11 + 11 = 33
const pwr = (a, b) => {
pwrCount++
return (b === 1) ? a : mpy(a, pwr(a, b - 1))
};
Out.log(mpy(11, 3)); // 33
Out.log(pwr(9, 4));
Out.log(secCount);
Out.log(addCount);
Out.log(mpyCount);
Out.log(pwrCount);
Out.log(pwr(9, 5));
